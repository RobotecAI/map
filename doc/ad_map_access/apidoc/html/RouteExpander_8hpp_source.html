185 template <class RoutingCostData>
186 void RouteExpander<RoutingCostData>::expandNeighbors(
187  typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
188 {
189  if (origin.first.direction == RoutingDirection::DONT_CARE)
190  {
191  throw std::runtime_error("RouteExpander::ExpandNeighbors invalid routing direction!");
192  }
193 
194  lane::Lane::ConstPtr lane = lane::getLanePtr(origin.first.point.laneId);
195  if (lane)
196  {
197  if ( // lane has to be routable to expand
198  isRouteable(*lane)
199  && ( // max distance and max duration are not yet reached
200  ((origin.second.routeDistance < mMaxDistance) && (origin.second.routeDuration < mMaxDuration))
201  // or we are within an intersection
202  || lane::isLanePartOfAnIntersection(*lane)))
203  {
204  expandSameLaneNeighbors(lane, origin);
205  expandLongitudinalNeighbors(lane, origin);
206  expandLateralNeighbors(lane, origin);
207  }
208  }
209  else
210  {
211  throw std::runtime_error("RouteExpander::ExpandNeighbors No lane!");
212  }
213 }

215 template <class RoutingCostData>
216 typename RouteExpander<RoutingCostData>::RoutingPoint
217 RouteExpander<RoutingCostData>::createNeighbor(lane::Lane::ConstPtr originLane,
218  RoutingPoint const &origin,
219  lane::Lane::ConstPtr neighborLane,
220  RoutingParaPoint neighborRoutingParaPoint)
221 {
222  RoutingPoint neighbor;
223  neighbor.first = neighborRoutingParaPoint;
224 
225  physics::Distance neighborDistance{0.};
226  physics::Duration neighborDuration{0.};
227  point::ECEFPoint pt_origin
228  = getParametricPoint(*originLane, origin.first.point.parametricOffset, physics::ParametricValue(0.5));
229  point::ECEFPoint pt_neighbor
230  = getParametricPoint(*neighborLane, neighbor.first.point.parametricOffset, physics::ParametricValue(0.5));
231  neighborDistance = point::distance(pt_neighbor, pt_origin);
232  physics::ParametricRange drivingRange;
233  if (origin.first.point.parametricOffset < neighbor.first.point.parametricOffset)
234  {
235  drivingRange.minimum = origin.first.point.parametricOffset;
236  drivingRange.maximum = neighbor.first.point.parametricOffset;
237  }
238  else
239  {
240  drivingRange.minimum = neighbor.first.point.parametricOffset;
241  drivingRange.maximum = origin.first.point.parametricOffset;
242  }
243 
244  if (originLane == neighborLane)
245  {
246  neighborDuration = getDuration(*originLane, drivingRange);
247  }
248  else
249  {
250  auto const maxSpeed = std::max(getMaxSpeed(*originLane, drivingRange), cMinimumSpeed);
251  neighborDuration = neighborDistance / maxSpeed;
252  }
253  neighborDistance = std::max(neighborDistance, cMinimumNeighborDistance);
254  neighborDuration = std::max(neighborDuration, cMinimumNeighborDuration);
255  neighbor.second.routeDistance = origin.second.routeDistance + neighborDistance;
256  neighbor.second.routeDuration = origin.second.routeDuration + neighborDuration;
257  return neighbor;
258 }

273 void RouteExpander<RoutingCostData>::expandSameLaneNeighbors(
274  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
275 {
276  if ((lane->id == getDest().laneId)
277  && ((isPositiveMovement(lane, origin) && (origin.first.point.parametricOffset <= getDest().parametricOffset))
278  || (isNegativeMovement(lane, origin) && (origin.first.point.parametricOffset >= getDest().parametricOffset))))
279  {
280  // approaching the destination from valid side
281  auto const neighbor = createNeighbor(lane, origin, lane, getRoutingDest());
282  addNeighbor(lane, origin, lane, neighbor, ExpandReason::Destination);
283  }
284  if (isPositiveMovement(lane, origin) && !isEnd(origin))
285  {
286  auto const neighbor = createNeighbor(
287  lane, origin, lane, createRoutingParaPoint(lane->id, physics::ParametricValue(1.), origin.first.direction));
288  addNeighbor(lane, origin, lane, neighbor, ExpandReason::SameLaneNeighbor);
289  }
290  if (isNegativeMovement(lane, origin) && !isStart(origin))
291  {
292  auto const neighbor = createNeighbor(
293  lane, origin, lane, createRoutingParaPoint(lane->id, physics::ParametricValue(0.), origin.first.direction));
294  addNeighbor(lane, origin, lane, neighbor, ExpandReason::SameLaneNeighbor);
295  }
296 }

298 template <class RoutingCostData>
299 void RouteExpander<RoutingCostData>::expandLongitudinalNeighbors(
300  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
301 {
302  lane::ContactLaneList contactLanes;
303  if (isEnd(origin) && isPositiveMovement(lane, origin))
304  {
305  contactLanes = lane::getContactLanes(*lane, lane::ContactLocation::SUCCESSOR);
306  }
307  else if (isStart(origin) && isNegativeMovement(lane, origin))
308  {
309  contactLanes = lane::getContactLanes(*lane, lane::ContactLocation::PREDECESSOR);
310  }
311  for (auto contactLane : contactLanes)
312  {
313  if (!isLaneRelevantForExpansion(contactLane.toLane))
314  {
315  continue;
316  }
317  lane::Lane::ConstPtr otherLane = lane::getLanePtr(contactLane.toLane);
318  if (otherLane)
319  {
320  if (isRouteable(*otherLane))
321  {
322  lane::ContactLocation otherToLane = getContactLocation(*otherLane, lane->id);
323  if (otherToLane == lane::ContactLocation::SUCCESSOR)
324  {
325  auto routingDirection = RoutingDirection::NEGATIVE;
326  auto const neighbor = createLongitudinalNeighbor(
327  origin, createRoutingParaPoint(otherLane->id, physics::ParametricValue(1.), routingDirection));
328 
329  addNeighbor(lane, origin, otherLane, neighbor, ExpandReason::LongitudinalNeighbor);
330  }
331  else if (otherToLane == lane::ContactLocation::PREDECESSOR)
332  {
333  auto routingDirection = RoutingDirection::POSITIVE;
334  auto const neighbor = createLongitudinalNeighbor(
335  origin, createRoutingParaPoint(otherLane->id, physics::ParametricValue(0.), routingDirection));
336 
337  addNeighbor(lane, origin, otherLane, neighbor, ExpandReason::LongitudinalNeighbor);
338  }
339  else
340  {
341  throw std::runtime_error("Other lane neither SUCCESSOR not PREDECESSOR!");
342  }
343  }
344  }
345  else
346  {
347  throw std::runtime_error("No other lane!");
348  }
349  }
350 }
